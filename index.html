Welcome to Zar

Zar is a fun (I hope), minimal scripting language with a retro style, built using Python. Why is it called ZAR? I reversed the word "ram" and changed the M to a Z because my nickname is Żurek (by the way, that's the name of a traditional Polish soup).

Features

  • Simple syntax
  • Built-in GUI support
  • Custom interpreter written in Python
  • File management (read/write)
  • Input/output operations
  • Looping and conditional control flow
  • Library support (e.g., time, os, platform)
  • Cross-platform screen clearing
  • GUI creation with CustomTkinter (windows, labels)
  • Basic calculator functionality

Basics
Variables 
  
// setting string variable "text" to "Some text"  
set str text : "Some text"  
// setting integer variable "number" to 17  
set int number : 17  
// setting integer variable "LengthOfText" to the length of variable "text"  
set int LengthOfText : $ text
"Hello, World!" Program 
  
set str msg : "Hello, World!"  
set int MsgLen : $ msg  
add MsgLen, 1  
write msg, MsgLen
Compare 
  
set str msg : "x == y!"  
set int MsgLen : $ msg  
add MsgLen, 1  
set int x : 1  
set int y : 1  
compare x | y  
write msg, MsgLen
What's Your Name (Input) 
  
set str ASK_MSG : "What is your name?"  
set int ASK_MSG_LEN : $ ASK_MSG  
set str msg : "name"  
write ASK_MSG, ASK_MSG_LEN  
read msg  
set str hello_msg : "Hello, "  
add hello_msg, msg  
add hello_msg, "!"  
set int MsgLen : $ hello_msg  
add MsgLen, 1  
write hello_msg, msg, MsgLen
Jump Loops 
  
set str msg : "Hello";  
set int count : 0;  
set int count_end : 3;  
  
loop:  
write msg, 5;  
add count, 1;  
compare count | count_end;  
jmp end;  
jmp loop;  
  
end:  
set str msg : "end";  
write msg, 3;
Making Lists 
set list test_list : [item1, item2]
Checking Lists Lenght 
set list test_list : [1, 2]
set int number : 2

set str msg : "list is 2 items long!"
set int MsgLen : $ msg

// compares lenght of list "test_list" with int variable "number"
// if lenght of the list (number of items) and int variable aren't the same
// it will skip next line so only if len(test_list) == number program will print str variable "msg"
compare.$list test_list | number
write msg, MsgLen

// numbers of items in lists starts from 0
//    0      1
// [item1, item2]
Checking Lists Items 
  
set list test_list : ["item1", "cat", "item3"]
set str item_name : "cat"

set str msg : "2nd word in list 'test_list' is 'cat'!"
set int MsgLen : $ msg

// skips next line if 1st item from list "test_list" isn't equal to variable "item_name"
//                list name item number    variable
compare.list.item test_list,     1       | item_name
write msg, MsgLen
Adding Items To Lists 
  
set list cool_list : ["item1", "item2"]
set int ListLen : 3

set str msg : "List lenght is 3!"
set int MsgLen : $ msg

// adding item "item3" to list "cool_list"
list.push cool_list, "item3"
compare.$list cool_list | ListLen
write msg, MsgLen
Deleting Items From Lists 
  
set list cool_list : ["item1", "item2", "item3"]
set int ListLen : 1

set str msg : "List lenght is 1!"
set int MsgLen : $ msg

// deleting item "item3" from list "cool_list"
list.pop cool_list, "item3"

// deleting item with index 0 from list "cool_list"
list.pop cool_list, 1

compare.$list cool_list | ListLen
write msg, MsgLen
Sub Items In Lists 
  
set list test_list : [[1, 2], [3, 4]]
set list sub_item : [1, 2]

set str msg : "1st sub item in list is [1, 2]"
set int MsgLen : $ msg

compare.list.item test_list, 0 | sub_item
write msg, MsgLen
Printing Lists 
  
set list CoolList : ["cat", "frog", "dog"]
write.list CoolList
File Management
Reading Files 

// setting variable msg to text from file "text file.txt"
set str msg : file("text file.txt")

set int MsgLen : $ msg
sub MsgLen, 1

write msg, MsgLen
Making Files 
// if file "folder/text file.txt" already exist then text in it will be changed
set str text : "new text"
file.new "folder/text file.txt" | text
Adding New Line 
// this code will add new line to file "cool file.py"
set str text : "print('Hello, World!')"
fileln "cool file.py" | text
Words Checking 
// if 2nd word in 3th line in file "test.txt" is "cat" program will print it else it will do nothing

// variables
set str msg : "2nd word in 3th line is 'cat'"
set str is_equal : "cat"
set int MsgLen : $ msg

//                 file name   line column   if var == word
file.words.compare test.txt  :  1     2    |   is_equal
write msg, MsgLen

OS

set str msg : "Hello 1"
set int MsgLen : $ msg
write msg, MsgLen

// clearing screen (works only on windows, use "clear" for linux)
cls

set str msg : "Hello 2"
set int MsgLen : $ msg
write msg, MsgLen
Windows+Linux "clear screen" Command 
// importing os and platform libraries
import os
import platform

set str msg : "Hello 1"
set int MsgLen : $ msg
write msg, MsgLen

// this command clears screen regardless of the operating system (windows/linux)
platform.system(clear screen)

set str msg : "Hello 2"
set int MsgLen : $ msg
write msg, MsgLen
pymsgbox
Making Message Boxes 
import pymsgbox

set str title : "Cool title"
set str text : "Clever text!"

pymsgbox.alert text | title
Getting Input 
import pymsgbox

set str Title : "Enter Something"
set str Message : "Please type your name:"

pymsgbox.prompt Title | Message | input

set str title : "Hello!"
set str text : "Hello "
set str face : " :D"

add text, input
add text, face

pymsgbox.alert text | title
OK/Cancel Boxes 
import pymsgbox

set str Title : ">:3"
set str Message : "Now Im gonna say 'cat' >:3"

pymsgbox.confirm Title | Message | input

compare input | "OK"
jmp ok
compare input | "Cancel"
jmp cancel
jmp end

ok:
    set str title : "cat"
    set str text : "cat"

    pymsgbox.alert text | title
    jmp end
cancel:
    set str title : "why?"
    set str text : ":("

    pymsgbox.alert text | title
    jmp end
end:
CustomTkinter
Making A Window 
import customtkinter

//           width height       title
window.create(400 | 300, "My First ZAR GUI")
window.mainloop
Making Label 
import customtkinter

set int x : 200
set int y : 200
set str msg : "Hello, World!"

window.create(400 | 400, "GUI Labels")

//          id  x   y  text
label.create(1, x | y, msg)

window.mainloop
Changing Texts Of Labels 
import customtkinter
// to use "label.change_after" you need to import time
import time

set int x : 100
set int y : 60
set str msg : "Waiting..."
set str msg2 : "Changed!"

window.create(400 | 300, "Auto Update Label")
label.create(1, x | y, msg)

// changing label id  text  after x time
label.change_after(1, msg2, 3)

window.mainloop
