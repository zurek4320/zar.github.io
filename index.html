      // messages
      set str MSG1 : "Select first number."
      set str MSG2 : "Select second number."
      set str MSG3 : "Select operator (+,-,*,/)"

      // length of messages
      set int MSG1LEN : $ MSG1
      set int MSG2LEN : $ MSG2
      set int MSG3LEN : $ MSG3

      // getting numbers and operator
      write MSG1, MSG1LEN
      read num1
      write MSG2, MSG2LEN
      read num2
      write MSG3, MSG3LEN
      read op

      set int num1 : /num1
      set int num2 : /num2

      // getting right section
      compare op | "+"
      jmp plus
      compare op | "-"
      jmp minus
      compare op | "*"
      jmp multiplication
      compare op | "/"
      jmp division
      jmp end

      plus:
          add num1, num2
          jmp end

      minus:
          sub num1, num2
          jmp end

      multiplication:
          mul num1, num2
          jmp end

      division:
          div num1, num2

      end:
          set str output : /num1
          set int len : $ output
          write output, len.
