Welcome to Zar

Zar is a fun (I hope), minimal scripting language with a retro style, built using Python. Why is it called ZAR? I reversed the word "ram" and changed the M to a Z because my nickname is Żurek (by the way, that's the name of a traditional Polish soup).

Features

  • Simple syntax
  • Built-in GUI support
  • Custom interpreter written in Python

Documentation

Variables

// setting string variable "text" to "Some text"
set str text : "Some text"
// setting integer variable "number" to 17
set int number : 17
// setting integer variable "LengthOfText" to the length of variable "text"
set int LengthOfText : $ text

"Hello, World!" program

// setting string variable "msg" to "Hello, World!"
set str msg : "Hello, World!"
// setting integer variable "MsgLen" to the length of variable "msg"
set int MsgLen : $ msg

// adding 1 to integer variable "MsgLen"
add MsgLen, 1

// writing value of string variable "msg"
write msg, MsgLen

// write arg1, arg2
// arg1 - text from variable
// arg2 (last argument) - length of the message written to the console

Compare

// the compare command works like a basic if statement:
// if var1 != var2, it skips one line of code
set str msg : "x == y!"
set int MsgLen : $ msg

add MsgLen, 1

set int x : 1
set int y : 1

// if x == y, the program will write "x == y!\n"

compare x | y
write msg, MsgLen

Input and "what's your name" program

set str ASK_MSG : "What is your name?"
set int ASK_MSG_LEN : $ ASK_MSG

set str msg : "name"

write ASK_MSG, ASK_MSG_LEN
read msg

set str hello_msg : "Hello, "

add hello_msg, msg
add hello_msg, "!"

set int MsgLen : $ hello_msg
add MsgLen, 1

write hello_msg, msg, MsgLen

Jmp loops

set str msg : "Hello";
set int count : 0;

// how much do jmp loop will take times to end
set int count_end : 3;

loop:
write msg, 5;

add count, 1;
compare count | count_end;

jmp end;
jmp loop;

end:
set str msg : "end";
write msg, 3;

Simple calculator app

// messages
set str MSG1 : "Select first number."
set str MSG2 : "Select second number."
set str MSG3 : "Select operator (+,-,*,/)"

// lengths of messages
set int MSG1LEN : $ MSG1
set int MSG2LEN : $ MSG2
set int MSG3LEN : $ MSG3

// getting numbers and operator
write MSG1, MSG1LEN
read num1
write MSG2, MSG2LEN
read num2
write MSG3, MSG3LEN
read op

set int num1 : /num1
set int num2 : /num2

// branching based on operator
compare op | "+"
jmp plus
compare op | "-"
jmp minus
compare op | "*"
jmp multiplication
compare op | "/"
jmp division
jmp end

plus:
    add num1, num2
    jmp end

minus:
    sub num1, num2
    jmp end

multiplication:
    mul num1, num2
    jmp end

division:
    div num1, num2

end:
    set str output : /num1
    set int len : $ output
    write output, len
